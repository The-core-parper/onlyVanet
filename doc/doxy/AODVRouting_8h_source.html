44  /*
45  * It implements a unique identifier for an arbitrary RREQ message
46  * in the network. See: rreqsArrivalTime.
47  */
48  class RREQIdentifier
49  {
50  public:
51  IPv4Address originatorAddr;
52  unsigned int rreqID;
53  RREQIdentifier(const IPv4Address& originatorAddr, unsigned int rreqID) : originatorAddr(originatorAddr), rreqID(rreqID) {};
54  bool operator==(const RREQIdentifier& other) const
55  {
56  return this->originatorAddr == other.originatorAddr && this->rreqID == other.rreqID;
57  }
58  };
59 
60  class RREQIdentifierCompare
61  {
62  public:
63  bool operator()(const RREQIdentifier& lhs, const RREQIdentifier& rhs) const
64  {
65  return lhs.rreqID < rhs.rreqID;
66  }
67  };
